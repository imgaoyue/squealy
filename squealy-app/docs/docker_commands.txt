Running Postgres
----------------
docker run --name adventureworks -d -ePOSTGRES_PASSWORD=secret -p5432:5432 adventureworks

Running MySQL Server
--------------------
docker run -eMYSQL_ROOT_PASSWORD=secret -eMYSQL_USER=adventureworks -eMYSQL_PASSWORD=adventureworks@123 -eMYSQL_DATABASE=adventureworks -p3306:3306 -d mysql

Running MS SQL Server
---------------------
docker run -e 'ACCEPT_EULA=Y' -e 'SA_PASSWORD=secret' -p 1433:1433 -d mcr.microsoft.com/mssql/server:2017-latest
